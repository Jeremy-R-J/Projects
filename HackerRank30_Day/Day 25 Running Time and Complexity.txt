import math

def isPrime(n):
    if n <= 1:
        return False
    sqr_n= math.sqrt(n)
    if sqr_n.is_integer():
        return False
    for i in range(2,int(sqr_n)+1):
        if n % i == 0:
            return False
    return True
num_test = int(input())
for i in range(num_test):
    n = int(input())
    if isPrime(n):
        print('Prime')
    else:
        print('Not prime')
